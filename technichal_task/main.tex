\section{Введение}
Цель данного документа — формализовать требования, архитектуру, сроки и критерии приёмки проекта комплексной системы для анализа шин по фото. Система состоит из трех основных сервисов:
\begin{enumerate}
	\item Измерение остаточной глубины протектора (раздел \ref{sec:depth_measurement})
	\item Подсчёт и классификация шипов на протекторе (раздел \ref{sec:spike_counting})
	\item Определение маркировки и размера шины по фото (раздел \ref{sec:tire_classification})
\end{enumerate}

\section{Измерение остаточной глубины протектора}
\label{sec:depth_measurement}

\subsection{Цели и задачи}
\begin{itemize}
	\item \textbf{Главная цель.} Определять глубину протектора шины по фотографии.
	\item \textbf{Ключевые метрики успеха (KPI).}
	      \begin{itemize}
		      \item Точность измерения глубины протектора:
		            \begin{itemize}
			            \item Среднее отклонение от истинного значения глубины: не более 0.6 мм.
			            \item Максимальное отклонение от исправленного значения глубины: не более 1 мм. (90\% случаев)
		            \end{itemize}
		      \item Время обработки одного запроса: не более 5 секунд (при наличии GPU) или 10 секунд (при отсутствии GPU). Данный пункт нуждается в проверке на реальных аппаратных средствах.
		      \item Доля отказов: не более 1\%.
		      \item Вывод ошибки при невозможности распознания (для повторного запроса с другой фотографией) в случае отказа системы.
	      \end{itemize}
\end{itemize}

\subsection{Функциональные требования}
\subsubsection{Входные данные}
\begin{itemize}
	\item Входные данные: изображение в формате JPEG, PNG.
	\item Содержание изображения:
	      \begin{itemize}
		      \item Единственная шина с четко видимым протектором.
		      \item Ориентация шины относительно камеры должна предоставлять прямой вид на протектор.
	      \end{itemize}
	\item Используется то же самое изображение, что и для подсчёта шипов на протекторе (см. раздел \ref{sec:spike_counting}).
\end{itemize}

\subsubsection{Выходные данные}
\begin{itemize}
	\item \textbf{Глубина протектора} в миллиметрах с округлением до сотых долей.
	\item Формат ответа: \texttt{JSON} с единственным полем \texttt{depth} (глубина протектора).
\end{itemize}

\section{Подсчёт и классификация шипов на протекторе}
\label{sec:spike_counting}

\subsection{Цели и задачи}
\begin{itemize}
	\item \textbf{Главная цель.} Подсчитать количество шипов на протекторе шины:
	      \begin{itemize}
		      \item Определить точки на изображении, соответствующие шипам.
		      \item Классифицировать каждый найденный шип на `хорошие' (полностью функциональные - 1 в json поле) и `плохие' (выбитые, сломанные, плавающие, замененные - 0 в json поле).
	      \end{itemize}
	\item \textbf{Ключевые метрики успеха (KPI).}
	      \begin{itemize}
		      \item Точность подсчёта количества шипов:
		            \begin{itemize}
			            \item Отклонение от истинного числа шипов на изображении: не более 10 штук (90\% случаев).
			            \item Отклонение от истинного числа `хороших' шипов: не более 7 штук (90\% случаев).
			            \item Отклонение от истинного числа `плохих' шипов: не более 7 штук (90\% случаев).
		            \end{itemize}
		      \item Время обработки одного запроса: не более 5 секунд (при наличии GPU) или 10 секунд (при отсутствии GPU). Данный пункт нуждается в проверке на реальных аппаратных средствах.
		      \item Доля отказов: не более 1\%.
		      \item Вывод ошибки при невозможности распознания (для повторного запроса с другой фотографией) в случае отказа системы.
	      \end{itemize}
\end{itemize}

\subsection{Функциональные требования}
\subsubsection{Входные данные}
\begin{itemize}
	\item Входные данные: изображение в формате JPEG, PNG.
	\item Содержание изображения:
	      \begin{itemize}
		      \item Единственная шина с четко видимым протектором.
		      \item Ориентация шины относительно камеры должна предоставлять прямой вид на протектор.
	      \end{itemize}
	\item Используется то же самое изображение, что и для измерения глубины протектора (см. раздел \ref{sec:depth_measurement}).
\end{itemize}

\subsubsection{Выходные данные}
\begin{itemize}
	\item \textbf{Шипы:} информация о каждом найденном шипе:
	      \begin{itemize}
		      \item \textbf{Классификация:} `хороший' или `плохой'.
		      \item \textbf{Координаты:} $[x, y, w, h]$ - координаты прямоугольника, в котором находится шип.
		      \item (Опционально) \textbf{Размеченное изображение:} визуализация результата обработки изображения для интерпретации и отладки.
	      \end{itemize}
	\item \textbf{Количество хороших шипов:} количество хороших шипов на изображении.
	\item \textbf{Количество плохих шипов:} количество плохих шипов на изображении.
	\item Формат ответа: \texttt{JSON} с полями \texttt{bad\_count}, \texttt{good\_count}, \texttt{spikes}.
\end{itemize}

\section{Определение маркировки и размера шины по фото}
\label{sec:tire_classification}
\subsection{Цели и задачи}
\begin{itemize}
	\item \textbf{Главная цель.} Принимать на вход изображение шины и выдавать:
	      \begin{itemize}
		      \item Марку шины.
		      \item Модель шины.
		      \item Размерность (текстовое описание, соответствующее стандарту).
	      \end{itemize}
	\item \textbf{Ключевые метрики успеха (KPI).}
	      \begin{itemize}
		      \item Точность OCR - правильное распознание маркировки и размера шины (не менее 90\% для каждого поля).
		      \item Время обработки одного запроса - не более 7 секунд.
		      \item Доля отказов - не более 1\%.
		      \item Вывод ошибки при невозможности распознания (для повторного запроса с другой фотографией) в случае отказа системы.
	      \end{itemize}
\end{itemize}

\subsection{Функциональные требования}

\subsubsection{Входные данные}
\begin{itemize}
	\item Входные данные: изображение в формате JPEG, PNG.
	\item Содержание изображения:
	      \begin{itemize}
		      \item Боковой вид единственной шины.
		      \item Четко различимые данные маркировки шины на резине.
	      \end{itemize}
\end{itemize}

\subsubsection{Выходные данные}
\begin{itemize}
	\item \textbf{Марка} и \textbf{модель} шины — значения из заранее определённого каталога категорий.
	\item \textbf{Размерность} шины в формате, определяемым заказчиком.
	\item Формат ответа: текстовое сообщение с необходимыми заказчику данными.
\end{itemize}

\section{Общие требования и условия}

\subsection{Нефункциональные требования}

\subsubsection{Производительность}
\begin{itemize}
	\item Время отклика: не более 10 секунд на обработку запроса.
	\item Пропускная способность: до 10 запросов в минуту (пик).
\end{itemize}

\subsubsection{Надёжность и доступность}
\begin{itemize}
	\item Доступность сервиса (SLA): не менее 99\% uptime.
	\item Гарантийная поддержка: 6 месяцев с момента ввода в эксплуатацию.
\end{itemize}

\subsubsection{Масштабируемость}
\begin{itemize}
	\item При необходимости масштабирования сервиса, продукт должен быть готов к вертикальному методу масштабируемости (увеличение вычислительной мощности на единственном сервере).
\end{itemize}

\subsubsection{Безопасность и приватность}
\begin{itemize}
	\item TLS 1.3 для всех соединений.
	\item Данные на диске не сохраняются (кроме кода и временных артефактов обработки).
	\item Отсутствие хранения пользовательских метаданных.
\end{itemize}

\subsubsection{Инфраструктура и контейнеризация}
\begin{itemize}
	\item Сервис упакован в Docker-контейнер, запущен на защищённом хостинге.
	\item Доступ к сервису осуществляется через REST API с аутентификацией по API-ключу.
\end{itemize}

\subsection{Тестирование и валидация}
\begin{itemize}
	\item Проверка корректности на валидационном наборе данных с известными метками при помощи тестов.
	\item Логирование каждого запроса: время, содержимое запроса, ответ сервиса.
\end{itemize}

\subsection{План внедрения и поддержки}

\subsubsection{Этапы разработки}
\begin{itemize}
	\item Очередность разработки сервисов будет определяться по мере поступления необходимых ресурсов и пожеланий заказчика.
\end{itemize}

\subsubsection{Гарантийная поддержка}
\begin{itemize}
	\item Сервис поддерживается разработчиком в течение 6 месяцев после ввода в эксплуатацию.
	\item В течение гарантийного срока - исправление ошибок, дополнительная валидация сервиса по запросу заказчика.
	\item В гарантийные случаи входят:
	      \begin{itemize}
		      \item Ошибки в коде или деплое сервиса, приводящие к помехам в работе сервиса.
		      \item Дообучение моделей на новых данных.
	      \end{itemize}
	\item В гарантийные случаи не входят:
	      \begin{itemize}
		      \item Добавление новых сервисов.
		      \item Доработка отдельных компонентов в целях улучшения качества работы сервиса.
		      \item Дополнительное тестирование и валидация сервиса.
	      \end{itemize}
\end{itemize}

	\item Случаи недостаточного качества работы сервиса обговариваются отдельно. Способы разрешения потенциальных проблем:
	      \begin{itemize}
		      \item Заказчик предоставляет дополнительные данные для дообучения моделей до требуемого уровня качества (входит в гарантийные случаи).
		      \item Спорные ситуации решаются проведением тестирования отдельных сервисов в целях выявления реального уровня качества на условиях, согласованных с заказчиком.
	      \end{itemize}
\end{itemize}
